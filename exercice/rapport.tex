\section{string\_of\_list}

\subsection{Implementation}

\begin{minted}{ocaml}
let string_of_list f l =
  let rec aux l = match l with
    | Nil -> "]"
    | Cons(a, Nil) -> f a^"]"
    | Cons (a,q) -> f a^", " ^ aux q
  in "["^aux l;;
\end{minted}
\subsection{Example}
\begin{minted}{ocaml}
string_of_list string_of_int l;;
\end{minted}
Sortie :
[1, 3, 6, 10 15, 21, 28, 36, 45, 55]
\section{hd}

\subsection{Implementation}

\begin{minted}{ocaml}
let hd l = match l with
  | Nil -> None
  | Cons (t,q) -> Some t;;
\end{minted}
\subsection{Example}
\begin{minted}{ocaml}
hd l
\end{minted}
Sortie :
Some 1
\section{tl}

\subsection{Implementation}

\begin{minted}{ocaml}
let tl l = match l with
  | Nil -> None
  | Cons (a,b) -> Some b;;
\end{minted}
\subsection{Example}
\begin{minted}{ocaml}
string_of_list string_of_int (Option.get (tl l))
\end{minted}
Sortie :
[3, 6, 10, 15, 21, 28, 36, 45, 55]
\section{length}

\subsection{Implementation}

\begin{minted}{ocaml}
let rec length l = match l with
  | Nil -> 0
  | Cons (a,b) -> 1 + length b;;
\end{minted}
\subsection{Example}
\begin{minted}{ocaml}
length l
\end{minted}
Sortie :
10
\section{map}

\subsection{Implementation}

\begin{minted}{ocaml}
let rec map f l = match l with
  | Nil -> Nil
  | Cons (a,b) -> Cons ( f a, map f b);;
\end{minted}
\subsection{Example}
\begin{minted}{ocaml}
string_of_liststring_of_int (map (fun n -> 2*n) l)
\end{minted}
